WarDuchyEffect = {
    root = {
        save_scope_as = war_attacker
    }
    
    faith = {
        great_holy_war = {
            every_pledged_attacker = {
                limit = {
                    exists = ghw_beneficiary
                    this = scope:war_attacker
                }
                ghw_beneficiary = {
                    save_scope_as = beneficiary
                }
            }
            ghw_war = {
                primary_defender = {
                    save_scope_as = defender
                    random_sub_realm_county = {
						limit = {
							duchy = {
								any_in_de_jure_hierarchy = {
									tier = tier_county
									holder = {
										OR = {
											THIS = scope:defender
											target_is_liege_or_above = scope:defender
										}
									}
									NOT  = { county_controller = scope:defender }
									exists = var:occupant
									var:occupant = scope:war_attacker
									count = all
								}
							}
						}
						duchy = {
							every_in_de_jure_hierarchy = {
								scope:beneficiary = {
									add_to_variable_list = {
										name = counties_transferred
										target = prev
									}
								}
							}
						}
						scope:beneficiary = {
							add_to_variable_list = {
								name = counties_transferred
								target = prev
							}
						}
					}
				}
			}
		}
	}


	scope:beneficiary = { 
		create_dynamic_title = {
			tier = kingdom
			name = "Test Crusader State"
		}
		create_title_and_vassal_change = {
			type = created
			save_scope_as = change
			add_claim_on_loss = no
		}
		scope:new_title = {
			set_no_automatic_claims = yes
			set_definitive_form = yes
			set_can_be_named_after_dynasty = no
			change_title_holder = {
				holder = scope:beneficiary 
				change = scope:change
			}
		}
		resolve_title_and_vassal_change = scope:change

		create_title_and_vassal_change = {
			type = conquest
			save_scope_as = change
			add_claim_on_loss = yes
		}

		every_in_list = {
			list = counties_transferred
			change_title_holder = {
				holder = scope:beneficiary
				change = scope:change
			}
		}

		resolve_title_and_vassal_change = scope:change

		create_title_and_vassal_change = {
			type = independency
			save_scope_as = change
			add_claim_on_loss = yes
		}
		becomes_independent = {
			change = scope:change
		}
		resolve_title_and_vassal_change = scope:change
	}
}
